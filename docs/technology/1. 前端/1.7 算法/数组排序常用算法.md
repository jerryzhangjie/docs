## 数组排序（专题整理 实现原理、时间复杂度、空间复杂度）

    let arr = [12, 2, 32, 13, 33, 5, 25]
    <!-- 冒泡排序 -->
    <!-- 
      1.比较相邻的两个元素，如果前一个比后一个大，则交换位置；
      2.比较完第一轮的时候，最后一个元素是最大的元素，无需参与之后的比较
     -->
    const len = arr.length - 1 // 假设3个数组，那么只需要2次就能完成排序，所以减1
    for (let i = 0; i < len; i++) {
      for (let j = 0; j < len - i; j++) {
        if (arr[j] > arr[j+1]) {
          let tem = arr[j]
          arr[j] = arr[j+1]
          arr[j+1] = tem
        }
      }
    }
